// Prisma schema for Freelancer Marketplace
// Datasource

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

enum Role {
  CLIENT
  FREELANCER
  ADMIN
}

enum JobStatus {
  OPEN
  IN_PROGRESS
  COMPLETED
  CANCELLED
}

enum ProposalStatus {
  PENDING
  ACCEPTED
  REJECTED
  WITHDRAWN
}

enum ContractStatus {
  ACTIVE
  COMPLETED
  CANCELLED
}

enum SupportStatus {
  OPEN
  IN_PROGRESS
  RESOLVED
}

enum NotificationType {
  SYSTEM
  JOB
  PROPOSAL
  CONTRACT
  REVIEW
  MESSAGE
}

enum PaymentStatus {
  PENDING
  PAID
  FAILED
  REFUNDED
}

enum TransactionType {
  DEBIT
  CREDIT
}

enum ReportStatus {
  OPEN
  IN_PROGRESS
  RESOLVED
  REJECTED
}

enum AdminActionType {
  BAN_USER
  UNBAN_USER
  DELETE_JOB
  DELETE_REVIEW
  RESOLVE_SUPPORT
  OTHER
}

enum ReportTargetType {
  USER
  JOB
  REVIEW
  MESSAGE
  CONTRACT
}

model User {
  id           String          @id @default(cuid())
  name         String
  email        String          @unique
  passwordHash String
  role         Role            @default(CLIENT)
  avatarUrl    String?
  isBanned     Boolean         @default(false)
  createdAt    DateTime        @default(now())
  updatedAt    DateTime        @updatedAt

  freelancerProfile FreelancerProfile?
  clientProfile     ClientProfile?

  jobs        Job[]            @relation("ClientJobs")
  proposals   Proposal[]       @relation("FreelancerProposals")
  contractsAsClient     Contract[] @relation("ClientContracts")
  contractsAsFreelancer Contract[] @relation("FreelancerContracts")

  sentMessages     Message[] @relation("SentMessages")
  receivedMessages Message[] @relation("ReceivedMessages")

  reviewsWritten Review[] @relation("ReviewsWritten")
  reviewsReceived Review[] @relation("ReviewsReceived")

  conversationsAsClient     Conversation[] @relation("ClientConversations")
  conversationsAsFreelancer Conversation[] @relation("FreelancerConversations")

  notifications Notification[]
  bookmarks     Bookmark[] @relation("BookmarkOwner")
  bookmarkedFreelancers Bookmark[] @relation("BookmarkFreelancer")
  supportTickets SupportTicket[]
  reports       Report[] @relation("ReportsByUser")
  adminActions  AdminAction[] @relation("AdminActionsByUser")

  transactions Transaction[]
  refreshTokens RefreshToken[]

  @@index([role])
  @@index([isBanned])
}

model FreelancerProfile {
  id            String   @id @default(cuid())
  userId        String   @unique
  title         String?
  bio           String?
  hourlyRate    Decimal? @db.Decimal(10, 2)
  rating        Float    @default(0)
  completedJobs Int      @default(0)
  totalIncome   Decimal  @default(0) @db.Decimal(12, 2)
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  user       User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  skills     FreelancerSkill[]
  portfolio  Portfolio[]

  @@index([rating])
}

model ClientProfile {
  id           String   @id @default(cuid())
  userId       String   @unique
  companyName  String?
  companySize  String?
  industry     String?
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)
}

model Job {
  id          String     @id @default(cuid())
  clientId    String
  title       String
  description String
  budget      Decimal    @db.Decimal(12, 2)
  location    String?
  workMode    String?
  experienceLevel String?
  status      JobStatus  @default(OPEN)
  categoryId  String?
  createdAt   DateTime   @default(now())
  updatedAt   DateTime   @updatedAt

  client    User      @relation("ClientJobs", fields: [clientId], references: [id], onDelete: Cascade)
  proposals Proposal[]
  contracts Contract[]
  category  Category? @relation(fields: [categoryId], references: [id])
  skills    JobSkill[]
  attachments Attachment[]

  @@index([clientId])
  @@index([status])
  @@index([categoryId])
}

model Proposal {
  id          String         @id @default(cuid())
  jobId       String
  freelancerId String
  coverLetter String?
  bidAmount   Decimal        @db.Decimal(12, 2)
  status      ProposalStatus @default(PENDING)
  createdAt   DateTime       @default(now())
  updatedAt   DateTime       @updatedAt

  job        Job  @relation(fields: [jobId], references: [id], onDelete: Cascade)
  freelancer User @relation("FreelancerProposals", fields: [freelancerId], references: [id], onDelete: Cascade)
  attachments Attachment[]

  @@index([jobId])
  @@index([freelancerId])
  @@index([status])
}

model Contract {
  id           String         @id @default(cuid())
  jobId        String
  clientId     String
  freelancerId String
  price        Decimal        @db.Decimal(12, 2)
  status       ContractStatus @default(ACTIVE)
  startedAt    DateTime       @default(now())
  completedAt  DateTime?
  createdAt    DateTime       @default(now())
  updatedAt    DateTime       @updatedAt

  job        Job  @relation(fields: [jobId], references: [id], onDelete: Cascade)
